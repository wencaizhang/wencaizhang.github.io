```bash
git remote -v # 查看远程仓库
```

```bash
git remote add origin # 添加远程仓库
```

```bash
git remote rm origin # 删除远程仓库
```

```bash
git remote set-url origin # 修改远程仓库
```

```bash
git remote show origin # 查看远程仓库详情
```

```bash
git remote rename origin newOrigin # 修改远程仓库名
```

```bash
git remote prune origin # 删除远程不存在的分支
```

```bash
git remote update origin --prune # 更新远程分支
```

```bash
git remote set-head origin master # 设置远程仓库的HEAD指向master分支
```

```bash
git remote set-head origin -d # 删除远程仓库的HEAD指向
```

```bash
git remote get-url origin # 获取远程仓库的URL
```

```bash
git remote set-url origin # 设置远程仓库的URL
```

```bash
git remote set-url --add origin # 添加远程仓库的URL
```
